    class Task {
        +String name
        +String category
        +int duration
        +String priority
        +String frequency
        +bool completed
        +Optional~date~ due_date
        +mark_complete() Optional~Task~
        +reset() void
        +is_due(date) bool
        +get_summary() String
    }

    class Pet {
        +String name
        +String species
        +String breed
        +int age
        +List~Task~ tasks
        +add_task(Task) void
        +remove_task(String) bool
        +mark_task_complete(String) Optional~Task~
        +get_pending_tasks() List~Task~
        +get_summary() String
    }

    class Owner {
        +String name
        +int available_minutes
        +List~Pet~ pets
        +add_pet(Pet) void
        +get_all_tasks() List~Task~
        +get_all_pending_tasks() List~Task~
        +get_summary() String
    }

    class DailyPlan {
        +List~Task~ scheduled_tasks
        +List~Task~ skipped_tasks
        +int total_time_used
        +display() String
        +get_reasoning() String
    }

    class Scheduler {
        +List~Task~ tasks
        +int available_minutes
        +sort_by_time(List~Task~) List~Task~
        +sort_by_priority(List~Task~) List~Task~
        +filter_by_status(bool) List~Task~
        +filter_by_pet(Owner, String) List~Task~
        +filter_by_category(String) List~Task~
        +detect_conflicts() List~String~
        +generate_plan() DailyPlan
    }
